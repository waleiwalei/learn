```js
// 递归
function find(arr, item, start, end) {
	if(start >= end) return -1;
	let middle = Math.floor((start + end)/2);
	if(arr[middle] === item) {
		return middle;
	} else if(arr[middle] < item) {
		return find(arr, item, middle + 1, end);
	} else {
		return find(arr, item, start, middle - 1);
	}
}

			end = middle;
		} else {
			start = middle;
		}
	}
	while(start <= end) {
		innerFind(arr, item, start, end);
	}
	return ret;
}
```

    + 快速排序
        > 选择基准位后，小于的放在左边，大于的放在右边，每次把基准元素的位置调整好
        ```js
        // 递归
        let arr = [19, 8, 1, 35, 12, 30];
        // [8, 1, 12] => [1] [8] [12]
        // [19]
        // [35, 30] [30] [35] []
        function quick(arr) {
            let cur = arr[0];
            let left = [], right = [];
            for(let i = 1; i < arr.length; i ++) {
                if(arr[i] < cur) {
                    left.push(arr[i]);
                } else {
                    right.push(arr[i]);
                }
            }
            return quick(left).concat([cur], quick(right));
        }

        // 非递归
        let arr = [19, 8, 1, 35, 12, 30];
        quick(arr);
        console.log(arr);
        function quick(arr) {
            _quick(arr, 0, arr.length - 1);
        }
        function _quick(arr, left, right) {
            let list = [[left, right]];
            while(list.length) {
                let curList = list.shift();
                let i = curList[0], j = curList[1];
                let cur = arr[curList[0]];
                while(i<j) {
                    while(i < j && arr[j] > cur) {
                        j--;
                    }
                    while(i < j && arr[i] <= cur) {
                        i++;
                    }
                    // 重合时交换cur与当前位置
                    if(i == j) {
                        let temp = arr[i];
                        arr[i] = arr[curList[0]];
                        arr[curList[0]] = temp;
                    } else {
                        let temp = arr[i];
                        arr[i] = arr[j];
                        arr[j] = temp;
                    }
                }
                // 如果左右两侧元素个数还多余1个 则继续
                if(i - curList[0] > 1) {
                    list.push([curList[0], i]);
                }
                if(curList[1] - i > 1) {
                    list.push([i + 1, curList[1]]);
                }
            }
        }
        ```

    + 归并排序
        > 归并：合并两个有顺序的数组
        > 排序：将数组无限对半折，从最基础的两个单一元素数组开始归并，循环出去 结果有序
        ```js
            let arr = [19, 8, 20, 4, 35, 13];
            let newArr = mergeSort(arr);
            function mergeSort(arr) {
                if(arr.length <= 1) return;
                let mid = parseInt(arr.length/2);
                let leftArr = arr.slice(0, mid);
                let rightArr = arr.slice(mid+1, arr.length);
                return merge(mergeSort(leftArr), mergeSort(rightArr));
            }
            function merge(left, right) {
                let arr = [];   // 合并后数组
                let i = 0, let j = 0;
                while(i < left.length && j < right.length) {
                    if(left[i] < right[j]) {
                        arr.push(left[i]);
                        i ++;
                    }
                    if(left[i] > right[j]) {
                        arr.push(right[j]);
                        j ++;
                    }
                }
                while(i < left.length) {
                    arr.push(left[i]);
                    i ++;
                }
                while(j < right.length) {
                    arr.push(right[j]);
                    j ++;
                }
                return arr;
            }
        ```

```js
/**
 * @param {string} text1
 * @param {string} text2
 * @return {number}
 */
var longestCommonSubsequence = function(t1, t2) {
    function getMaxNum(text1, text2){
        let obj1 = {}, obj2 = {};
        let retArr = [], retArrMax = -1;
        for(let i = 0; i < text1.length; i ++) {
            obj1[text1[i]] = i;
        }
        for(let i = 0; i < text2.length; i ++) {
            obj2[text2[i]] = i;
        }
        let keys = Object.keys(obj1);
        for(let i = 0; i < keys.length; i ++) {
            let chart = keys[i];
            if(obj2[chart] >= 0 && retArrMax < obj2[chart]) {
                retArr.push(obj2[chart]);
                retArrMax = obj2[chart];
            }
        }
        return retArr.length;
    }
    let num1 = getMaxNum(t1, t2);
    let num2 = getMaxNum(t2, t1);
    return num1 > num2 ? num1 : num2;
};
```

- 一个数字三角形，每个节点只能向他的下面[i+1][j]或右下面[i+1][j+1]走，求最大路径值
    > 动态规划 状态转移方程
    ```js
    var arr = [
        [5], 
        [6,  7],
        [9,  10,  3],
        [12, 23,  0,  20],
        [8,  10,  11, 16, 8]
    ];
    // 递归
    // 会有重复的计算过程
    function findMax(arr, i, j) {
        if(!arr[i][j] || i == arr.length - 1) return arr[i][j] || 0;
        return arr[i][j] + Math.max(
            findMax(arr, i + 1, j),
            findMax(arr, i + 1, j + 1)
        )
    }
    let maxSum = [];
    function findMax(arr, i, j) {
        if(!arr[i][j] || i == arr.length - 1) return arr[i][j] || 0;
        if(maxSum[i][j]) {
            return maxSum[i][j];
        } else {
            maxSum[i][j] = arr[i][j] + Math.max(
                findMax(arr, i + 1, j),
                findMax(arr, i + 1, j + 1)
            )
        }
        return maxSum[i][j];
    }
    // 非递归
    function findMax(arr) {
        for(let i = arr.length - 2; i >= 0; i --) {
            for(let j = 0; j < arr[i].length; j ++) {
                return arr[i][j] + Math.max(
                    arr[i+1][j], arr[i+1][j+1]
                )
            }
        }
        return arr[0][0];
    }
```
